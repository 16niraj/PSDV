def fib(n):
    f1=0
    f2=1
    l=[0,1]
    for i in range(3,n+1):
        fi=f2+f1
        l.append(fi)
        f1=f2
        f2=fi
    return l
print(fib(6))

import matplotlib.pyplot as mp

# xpoints=[0,1,2,3,4,5]
mp.plot(fib(6))
mp.show()
